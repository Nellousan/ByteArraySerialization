286  inline void constructFromPayload(const char* data)
287  {
288  size_t size = 0;
289  for (size_t i = 0; i < _BAS_CHECKSUM_SIZE_; i++) {
290  size |= (data[i] << (i * 8));
291  }
292  _data.assign(data, data + size + _BAS_CHECKSUM_SIZE_);
293  }

295  inline void checksumUpdate(void)
296  {

298 
299  if (_isChecksumRemoved)
300  addChecksum();

307  inline void pushRawData(size_t size, size_t array_size, const char* data)
308  {
309  for (size_t j = 0; j < array_size; j++) {
310  for (size_t i = 0; i < size; i++) {
311  _data.push_back(data[j * size + i]);
312  }
313  }

316  inline void pushSizes(size_t size, size_t array_size)
317  {
318  for (size_t i = 0; i < _BAS_SIZE_BYTES_; i++)
319  _data.push_back((size >> (i * 8)) & 0xFF);
320 
321  for (size_t i = 0; i < _BAS_ARRAY_SIZE_; i++)
322  _data.push_back((array_size >> (i * 8)) & 0xFF);
323  }

325  inline std::tuple<int, int> getSizes(void)
326  {
327  size_t size = 0, array_size = 0;
328 
329  if (!_isChecksumRemoved) {
330  removeChecksum();
331  }
332 
333  for (size_t i = 0; i < _BAS_SIZE_BYTES_; i++) {
334  size |= (_data.front() << (i * 8));
335  _data.erase(_data.begin());
336  }
337 
338  for (size_t i = 0; i < _BAS_ARRAY_SIZE_; i++) {
339  array_size |= (_data.front() << (i * 8));
340  _data.erase(_data.begin());
341  }
342  return std::make_tuple(size, array_size);
343  }

347  {
348  for (size_t j = 0; j < array_size; j++) {

350  for (size_t i = 0; i < size; i++) {
351  _data.erase(_data.begin());
